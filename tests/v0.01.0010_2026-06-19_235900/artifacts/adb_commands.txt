# v0.01.0010 真机测试 — 主要 adb 命令记录
# 设备: OnePlus 7 Pro (7fbf2094)
# 日期: 2026-06-19

# 设备列表
adb devices -l

# 安装 Debug APK
adb install -r ChinesePinyinIME\app\build\outputs\apk\debug\app-debug.apk

# 启动应用（验证版本页）
adb shell am start -n com.mercury.chinesepinyinime/.MainActivity
adb shell uiautomator dump /sdcard/ime_ui.xml
adb pull /sdcard/ime_ui.xml

# 打开短信编写界面
adb shell am start -a android.intent.action.SENDTO -d sms:10086

# 检查 IME 状态
adb shell dumpsys input_method | findstr /i "mInputShown mCurMethodId mercury"

# IME 窗口位置（用于坐标校准）
adb shell dumpsys window windows | findstr /i "InputMethod mFrame"

# 截屏
adb shell screencap -p /sdcard/test.png
adb pull /sdcard/test.png

# 模拟点击（示例：底栏 123 键，坐标需按设备校准）
adb shell input tap 123 2990
