# ADB commands used for v0.01.0021 test session (9-key T9 pinyin prototype)
$ADB = $LOCALAPPDATA/Android/Sdk/platform-tools/adb.exe

adb devices -l
adb install -r ChinesePinyinIME/app/build/outputs/apk/debug/app-debug.apk
adb shell am start -n com.mercury.chinesepinyinime/.MainActivity
adb shell uiautomator dump /sdcard/X.xml && adb pull /sdcard/X.xml <dest>   # used to read exact button bounds on the Activity (settings page views ARE visible to uiautomator, unlike IME keyboard buttons)
adb shell am start -a android.intent.action.SENDTO -d sms:
adb shell input tap <x> <y>
adb shell input swipe <x> <y> <x> <y> <duration_ms>   # long-press simulation for DEL repeat-delete
adb shell screencap -p /sdcard/X.png && adb pull /sdcard/X.png <dest>
adb shell dumpsys window | grep -i "InputMethod"   # check mInputShown / IME window frame
adb shell dumpsys input_method | grep -i mInputShown
adb logcat -c ; adb logcat -d -b crash   # caught the static-initializer crash bug before it shipped
adb shell am force-stop com.android.mms ; reopen sms: intent   # workaround when IME window failed to reattach after `adb install -r` replaced the running IME's APK
